	private string checkIPlink(string IP)
	{
		string result = "";
		Ping ping = new Ping();
		string hostNameOrAddress = IP.ToString();
		PingReply pingReply = ping.Send(hostNameOrAddress);
		if (pingReply.Status == IPStatus.Success)
		{
			result = "success";
		}
		return result;
	}
